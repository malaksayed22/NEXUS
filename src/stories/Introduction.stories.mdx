# MyUI

A portfolio-grade React component library for product teams that care about craft, accessibility, and consistent interface behavior.

## Getting Started

```bash
npm install
npm run dev
npm run storybook
```

```tsx
import { ToastProvider, Card, Tooltip, Button } from "myui/src";

export const App = () => (
  <ToastProvider>
    <Card variant="elevated" padding="md">
      <Card.Body>
        <Tooltip content="Launch your workflow">
          <Button variant="primary" size="md">
            Launch
          </Button>
        </Tooltip>
      </Card.Body>
    </Card>
  </ToastProvider>
);
```
